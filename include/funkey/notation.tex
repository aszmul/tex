\usepackage{xifthen}

% isin version that expands its first argument
\newcommand*{\isinxp}[2]{\expandafter\isin\expandafter{#1}{#2}}

\newcommand{\mbb}[1]{{%
\ifthenelse{\isinxp{#1}{abcdefghijklmnopqrstuvxyz}}%
  {{\mathbf #1}}%
  {\boldsymbol #1}%
}}
\newcommand{\mup}[1]{{%
\ifthenelse{\isin{#1}{abcdefghijklmnopqrstuvxyzABCDEFGHIJKLMNOPQRSTUVXYZ}}%
  {\text{#1}}%
  {#1}%
}}

% vector
\newcommand{\vct}[1]{\protect\mbb{#1}}
% matrix
\newcommand{\mat}[1]{#1}
% set
\newcommand{\set}[1]{\protect\mbb{#1}}
% super-set
\newcommand{\sset}[1]{\boldsymbol{\mathcal{#1}}}
% random variable
\newcommand{\ran}[1]{#1}
% function
\newcommand{\fun}[1]{\mathrm{#1}}

% whole numbers
\newcommand{\whole}[0]{\mathbb{Z}}
% real numbers
\newcommand{\real}[0]{\mathbb{R}}
% natural numbers
\renewcommand{\natural}[0]{\mathbb{N}}
